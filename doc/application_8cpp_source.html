208  std::array<int16_t, 3> normalise(const glm::vec3& normal)
209  {
210  std::array<int16_t, 3> result;
211 
212  if (normal.x == 1.0) result.at(0) = INT16_MAX;
213  else if (normal.x == -1.0) result.at(0) = INT16_MIN;
214  else result.at(0) = static_cast<int16_t>(normal.x * static_cast<float>(INT16_MAX));
215 
216  if (normal.y == 1.0) result.at(1) = INT16_MAX;
217  else if (normal.y == -1.0) result.at(1) = INT16_MIN;
218  else result.at(1) = static_cast<int16_t>(normal.y * static_cast<float>(INT16_MAX));
219 
220  if (normal.z == 1.0) result.at(2) = INT16_MAX;
221  else if (normal.z == -1.0) result.at(2) = INT16_MIN;
222  else result.at(2) = static_cast<int16_t>(normal.z * static_cast<float>(INT16_MAX));
223 
224  return result;
225  }
